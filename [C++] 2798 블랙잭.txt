#include <iostream>
using namespace std;

int main() {
	int n, m,*arr;
	cin >> n >> m;
	arr = new int[n];
	int max = 0,sum;
	for (int i = 0;i < n;i++) cin >> arr[i];
	for (int i = 0;i < n-2;i++)
		for (int j = i+1;j < n-1;j++)
			for (int k = j + 1;k < n;k++) {
				sum = arr[i] + arr[j] + arr[k];
				if (max < sum && sum <= m) max = sum;
			}
	printf("%d", max);
	return 0;
}
