typedef long long ll;
using namespace std;
int arr[1001][1001];
string brr[1001][1001];
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	string s, f;
	cin >> f;
	cin >> s;
	int a = s.size(), b = f.size();
	memset(arr, 0, sizeof(arr));
	for(int i=1;i<=a;i++)
		for (int j = 1;j <= b;j++) 
			if (s[i - 1] == f[j - 1]) {
				arr[i][j] = arr[i - 1][j - 1] + 1;
				brr[i][j] = brr[i - 1][j - 1] + s[i - 1];
			}
			else {
				if (arr[i - 1][j] > arr[i][j - 1]) {
					arr[i][j] = arr[i - 1][j];
					brr[i][j] = brr[i - 1][j];
				}
				else {
					arr[i][j] = arr[i][j-1];
					brr[i][j] = brr[i][j-1];

				}
			}
	cout << arr[a][b] << '\n' << brr[a][b];
	return 0;
}
