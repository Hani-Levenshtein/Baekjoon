#include <iostream>
#include <algorithm>
#include <queue>
#include <string.h>
#include <limits.h>
typedef long long ll;
using namespace std;

int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int n,m;
	cin >> n;
	priority_queue <int,vector<int>,greater<int>> pq;
	while (n--) {
		cin >> m;
		if (m != 0) pq.push(m);
		else {
			if (pq.empty() == true)cout << "0" << '\n';
			else {
				cout << pq.top()<<'\n';
				pq.pop();
			}

		}
	}
	return 0;
}
